function halfSearch(arr, target) {
  let low = 0
  let high = arr.length
  let index = 0

  while (low <= high) {
    index = Math.floor((low + high) / 2)
    const middleValue = arr[index]

    if (middleValue === target) {
      return index
    } else if (middleValue > target) {
      high = index - 1
    } else {
      low = index + 1
    }
  }

  return -1
}
